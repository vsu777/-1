        // --- FootprintMap ---

        const WIDTH = 800;

        const HEIGHT = 600;

        const CENTER_LON = 105;

        const CENTER_LAT = 35;

        const SCALE = 10;

 

        const project = (lon, lat) => {

            const x = (lon - CENTER_LON) * SCALE * 1.2 + WIDTH / 2;

            const y = HEIGHT / 2 - (lat - CENTER_LAT) * SCALE * 1.4;

            return [x, y];

        };

 

        const FootprintMap = forwardRef(({ visitedCities, onError, setLoading }, ref) => {

            const [geoData, setGeoData] = useState(null);

            const svgRef = useRef(null);

            const [transform, setTransform] = useState({ k: 1, x: 0, y: 0 });

            const [isDragging, setIsDragging] = useState(false);

            const [startPoint, setStartPoint] = useState({ x: 0, y: 0 });

 

            useEffect(() => {

                setLoading(true);

                fetchChinaGeoJSON()

                    .then(data => setGeoData(data))

                    .catch(err => onError("Failed to load map data. Please try again."))

                    .finally(() => setLoading(false));

            }, [setLoading, onError]);

 

            useImperativeHandle(ref, () => ({

                downloadScreenshot: () => {

                    if (!svgRef.current) return;

                    const svgData = new XMLSerializer().serializeToString(svgRef.current);

                    const canvas = document.createElement("canvas");

                    canvas.width = WIDTH;

                    canvas.height = HEIGHT;

                    const ctx = canvas.getContext("2d");

                    const img = new Image();

                    img.onload = () => {

                        if (ctx) {

                            ctx.fillStyle = "#111827";

                            ctx.fillRect(0, 0, WIDTH, HEIGHT);

                            ctx.drawImage(img, 0, 0);

                            const link = document.createElement('a');

                            link.href = canvas.toDataURL("image/png");

                            link.download = `footprint-map-${new Date().toISOString().slice(0, 10)}.png`;

                            link.click();

                        }

                    };

                    img.src = "data:image/svg+xml;base64," + btoa(unescape(encodeURIComponent(svgData)));

                }

            }));

 

            const generatePath = (feature) => {

                const coords = feature.geometry.coordinates;

                const type = feature.geometry.type;

                const drawRing = (ring) => {

                    return ring.map((pt, i) => {

                        const [x, y] = project(pt[0], pt[1]);

                        return `${i === 0 ? 'M' : 'L'}${x.toFixed(1)},${y.toFixed(1)}`;

                    }).join(' ') + 'Z';

                };

                if (type === 'Polygon') return coords.map(ring => drawRing(ring)).join(' ');

                else if (type === 'MultiPolygon') return coords.map(poly => poly.map(ring => drawRing(ring)).join(' ')).join(' ');

                return '';

            };

 

            const visitedProvinces = useMemo(() => {

                const set = new Set();

                visitedCities.forEach(city => {

                    if (city.province) set.add(city.province);

                    if (city.type === 'province') set.add(city.name);

                });

                return set;

            }, [visitedCities]);

 

            const handleWheel = (e) => {

                e.preventDefault();

                const scaleFactor = 1.1;

                const direction = e.deltaY < 0 ? 1 : -1;

                let newK = direction > 0 ? transform.k * scaleFactor : transform.k / scaleFactor;

                newK = Math.max(1, Math.min(newK, 8));

                setTransform(prev => ({ ...prev, k: newK }));

            };

 

            const handleMouseDown = (e) => {

                setIsDragging(true);

                setStartPoint({ x: e.clientX - transform.x, y: e.clientY - transform.y });

            };

 

            const handleMouseMove = (e) => {

                if (!isDragging) return;

                setTransform(prev => ({ ...prev, x: e.clientX - startPoint.x, y: e.clientY - startPoint.y }));

            };

 

            const handleMouseUp = () => setIsDragging(false);

 

            const showLabels = transform.k > 2.0;

 

            return (

                <div className="w-full h-full relative flex justify-center items-center overflow-hidden bg-gray-900">

                    {geoData ? (

                        <svg

                            ref={svgRef}

                            viewBox={`0 0 ${WIDTH} ${HEIGHT}`}

                            className="w-full h-full max-w-5xl drop-shadow-2xl cursor-move"

                            style={{ fontFamily: 'sans-serif', touchAction: 'none' }}

                            preserveAspectRatio="xMidYMid meet"

                            onWheel={handleWheel}

                            onMouseDown={handleMouseDown}

                            onMouseMove={handleMouseMove}

                            onMouseUp={handleMouseUp}

                            onMouseLeave={handleMouseUp}

                        >

                            <defs>

                                <filter id="glow" x="-20%" y="-20%" width="140%" height="140%">

                                    <feGaussianBlur stdDeviation="3" result="blur" />

                                    <feComposite in="SourceGraphic" in2="blur" operator="over" />

                                </filter>

                            </defs>

                            <g transform={`translate(${transform.x}, ${transform.y}) scale(${transform.k})`} style={{ transformOrigin: 'center' }}>

                                {geoData.features.map((feature, i) => {

                                    const geoName = feature.properties.name;

                                    const isVisited = Array.from(visitedProvinces).some(vp => geoName.includes(vp));

                                    return (

                                        <path

                                            key={feature.properties.adcode || i}

                                            d={generatePath(feature)}

                                            fill={isVisited ? POLYGON_FILL_COLOR : "#1f2937"}

                                            stroke={POLYGON_STROKE_COLOR}

                                            strokeWidth={isVisited ? (0.8 / transform.k) : (0.3 / transform.k)}

                                            strokeOpacity={0.8}

                                            fillOpacity={isVisited ? 0.8 : 0.5}

                                            className="transition-colors duration-300 hover:fill-teal-600/50"

                                            vectorEffect="non-scaling-stroke"

                                        />

                                    );

                                })}

                                {visitedCities.filter(c => c.type === 'city').map((city, i) => {

                                    if (!city.coordinates) return null;

                                    const [cx, cy] = project(city.coordinates[0], city.coordinates[1]);

                                    const dotSize = 2 / transform.k + 1;

                                    return (

                                        <g key={city.name + i} className="group pointer-events-none">

                                            <circle cx={cx} cy={cy} r={dotSize} fill="#fbbf24" />

                                            <circle cx={cx} cy={cy} r={dotSize * 3} fill="#fbbf24" fillOpacity="0.3" />

                                            {showLabels && (

                                                <text

                                                    x={cx} y={cy - (10 / transform.k)}

                                                    textAnchor="middle"

                                                    fill="white"

                                                    fontSize={Math.max(8, 12 / transform.k)}

                                                    fontWeight="bold"

                                                    className="drop-shadow-md"

                                                >

                                                    {city.name}

                                                </text>

                                            )}

                                        </g>

                                    );

                                })}

                            </g>

                        </svg>

                    ) : (

                        <div className="text-teal-500 animate-pulse font-mono">Loading Map Data...</div>

                    )}

                    <div className="absolute top-4 right-4 bg-black/50 text-white/50 text-xs px-2 py-1 rounded pointer-events-none select-none">

                        Scroll to Zoom • Drag to Pan

                    </div>

                </div>

            );

        });

 

        // --- App ---

        const App = () => {

            const [visitedCities, setVisitedCities] = useState([]);

            const [loading, setLoading] = useState(false);

            const [errorMsg, setErrorMsg] = useState(null);

            const mapRef = useRef(null);

 

            useEffect(() => {

                const saved = localStorage.getItem(LOCAL_STORAGE_KEY);

                if (saved) {

                    try {

                        setVisitedCities(JSON.parse(saved));

                    } catch (e) {

                        console.error("Failed to parse saved cities", e);

                    }

                }

            }, []);

 

            useEffect(() => {

                localStorage.setItem(LOCAL_STORAGE_KEY, JSON.stringify(visitedCities));

            }, [visitedCities]);

 

            const handleAddCity = useCallback(async (cityName) => {

                if (visitedCities.some(c => c.name === cityName)) {

                    setErrorMsg(`City ${cityName} is already added!`);

                    setTimeout(() => setErrorMsg(null), 3000);

                    return;

                }

                setLoading(true);

                try {

                    const cityData = await searchCityLocation(cityName);

                    setVisitedCities(prev => {

                        if (prev.find(p => p.name === cityData.name)) return prev;

                        return [...prev, cityData];

                    });

                } catch (err) {

                    setErrorMsg(err.message || "Location not found");

                    setTimeout(() => setErrorMsg(null), 3000);

                } finally {

                    setLoading(false);

                }

            }, [visitedCities]);

 

            const handleClear = useCallback(() => {

                if (window.confirm("Are you sure you want to clear your travel history?")) {

                    setVisitedCities([]);

                }

            }, []);

 

            const handleDownload = () => {

                if (mapRef.current) mapRef.current.downloadScreenshot();

            };

 

            return (

                <div className="relative w-full h-screen flex flex-col bg-gray-900 overflow-hidden font-sans">

                    {errorMsg && (

                        <div className="absolute top-24 left-1/2 transform -translate-x-1/2 z-50 bg-red-500 text-white px-6 py-3 rounded-full shadow-xl animate-bounce">

                            {errorMsg}

                        </div>

                    )}

                    <StatsCard visitedCities={visitedCities} />

                    <div className="flex-1 relative z-0 pb-72">

                        <FootprintMap

                            ref={mapRef}

                            visitedCities={visitedCities}

                            onCityAdded={() => {}}

                            onError={(msg) => {

                                setErrorMsg(msg);

                                setTimeout(() => setErrorMsg(null), 4000);

                            }}

                            setLoading={setLoading}

                        />

                    </div>

                    <button

                        onClick={handleDownload}

                        className="fixed bottom-80 right-6 z-30 bg-white text-teal-600 px-4 py-3 rounded-full shadow-xl flex items-center justify-center space-x-2 hover:bg-teal-50 hover:scale-105 active:scale-95 transition-all group font-medium"

                        title="Download Map Image"

                    >

                        <Download className="w-5 h-5 group-hover:animate-bounce" />

                        <span>Save Map</span>

                    </button>

                    <ControlPanel onAddCity={handleAddCity} onClear={handleClear} isLoading={loading} />

                </div>

            );

        };

 

        // --- Render ---

        const root = createRoot(document.getElementById('root'));

        root.render(<App />);

    </script>
